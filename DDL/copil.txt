CREATE TABLE "COPIL"(
    "NUME" VARCHAR2(50) NOT NULL,
    "VARSTA" NUMBER(3),
    "SCOALA/UNIV" VARCHAR2(70),
    "SIT_MED" VARCHAR2(70),
    "ID_FAM" NUMBER(4),
    PRIMARY KEY("NUME") USING INDEX ENABLE
);

Constraint ID_FAM foreign key:

ALTER TABLE "COPIL" ADD CONSTRAINT "IDFAM_CON_COPIL" FOREIGN KEY ("ID_FAM")
	  REFERENCES "FAMILIE" ("ID_FAM") ON DELETE CASCADE ENABLE;


Constraint SCOALA/UNIV:
ALTER TABLE "COPIL"
ADD CONSTRAINT chk_scoalauniv 
CHECK (
    (VARSTA < 18 AND "SCOALA/UNIV" NOT LIKE 'Universitatea%') 
    OR VARSTA >= 18
);


Constraint SIT_MED:
ALTER TABLE "COPIL"
ADD CONSTRAINT check_sitmed_format_copil
CHECK (REGEXP_LIKE(SIT_MED, '^(NU ARE PROBLEME|[A-Za-z ]+)$') AND NOT REGEXP_LIKE(SIT_MED, '^NU ARE PROBLEME .+'));

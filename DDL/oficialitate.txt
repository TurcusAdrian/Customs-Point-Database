CREATE TABLE "OFICIALITATE"(
    "NUME" VARCHAR2(50) NOT NULL,
    "FUNCTIE" VARCHAR2(30) NOT NULL,
    "TARA" VARCHAR2(30),
    "ID_PV" NUMBER(4),
    "ID_MASINA" NUMBER(4),
    PRIMARY KEY("NUME") USING INDEX ENABLE
);

ALTER TABLE "OFICIALITATE" ADD CONSTRAINT "IDPV_CON_OFICIALITATE" FOREIGN KEY ("ID_PV")
	  REFERENCES "PUNCT VAMAL" ("ID_PV") ON DELETE CASCADE ENABLE;


Constraint "ID_MASINA" foreign key:
